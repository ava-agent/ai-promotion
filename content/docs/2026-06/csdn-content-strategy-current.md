```python
def auto_check_cdp_port():
    """每5分钟检查一次CDP端口状态"""
    if not port_is_open(9222):
        # 1. 记录告警
        log_alert("CDP端口异常，尝试自动修复...")
        # 2. 自动尝试修复
        kill_all_chrome()
        start_chrome_with_cdp()
        # 3. 二次验证
        if not port_is_open(9222):
            # 4. 修复失败，发送告警给主人
            send_alert_to_master("CDP自动修复失败，请手动处理")
```
