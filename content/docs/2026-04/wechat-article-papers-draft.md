```java
public class KnowledgeGraph {
    private Map<String, Article> articleMap;
    private Graph<String> citationGraph;
    
    public void addArticle(Article article) {
        articleMap.put(article.getId(), article);
        
        // 自动建立文章间的引用关系
        for (String reference : article.getReferences()) {
            citationGraph.addEdge(article.getId(), reference);
        }
    }
    
    public List<Article> findRelatedArticles(String articleId) {
        return citationGraph.getNeighbors(articleId).stream()
            .map(articleMap::get)
            .collect(Collectors.toList());
    }
}
```
